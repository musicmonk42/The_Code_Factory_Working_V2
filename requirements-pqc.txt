# Optional Post-Quantum Cryptography (PQC) Dependencies
# =======================================================
#
# These packages are NOT required for basic platform operation.
# Install only if you need quantum-resistant signing for audit logs.
#
# To enable PQC signing:
#   1. Install one of the libraries below:
#        pip install -r requirements-pqc.txt
#   2. Set the environment variable:
#        USE_PQC_SIGNING=true
#
# If neither library is installed but USE_PQC_SIGNING=true, the platform
# falls back to HMAC-SHA-256 and logs a warning.

# CRYSTALS-Dilithium via the pqcrypto binding (NIST PQC finalist)
# pqcrypto>=0.1.3

# Open Quantum Safe (OQS) Python binding — supports Dilithium3, Kyber, etc.
# Requires the liboqs C library to be installed first:
#   apt-get install -y cmake ninja-build libssl-dev
#   git clone --depth 1 https://github.com/open-quantum-safe/liboqs.git
#   cmake -S liboqs -B liboqs/build -DBUILD_SHARED_LIBS=ON && cmake --build liboqs/build --parallel && sudo cmake --install liboqs/build
# liboqs-python>=0.12.0
